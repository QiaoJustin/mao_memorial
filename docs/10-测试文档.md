## 1. 测试概述

### 1.1 测试目标

确保毛主席生平纪念网站在功能、性能、安全、兼容性等方面满足需求文档中定义的各项指标，保证系统稳定可靠地上线运行。

### 1.2 测试范围

| 测试类型 | 范围 | 工具 |
|---------|------|------|
| 单元测试 | 工具函数、组件逻辑、API 处理函数 | Vitest |
| 集成测试 | API 与数据库交互、模块间协作 | Vitest + Supertest |
| E2E 测试 | 关键用户流程 | Playwright |
| 性能测试 | 并发、响应时间、吞吐量 | k6 / Apache JMeter |
| 兼容性测试 | 浏览器、设备、分辨率 | BrowserStack |
| 安全测试 | XSS、CSRF、SQL注入、认证 | OWASP ZAP |
| 可访问性测试 | WCAG 2.1 AA | axe-core |

### 1.3 测试阶段

| 阶段 | 内容 | 时机 |
|------|------|------|
| 单元测试 | 代码级测试 | 开发阶段 |
| 集成测试 | 模块级测试 | 开发完成 |
| 系统测试 | 全功能测试 | 集成完成 |
| 验收测试 | 用户验收 | 上线前 |
| 回归测试 | 修改后验证 | 每次修改后 |

```typescript
import { describe, it, expect, beforeEach, vi } from 'vitest';
import { GET } from '@/app/api/v1/timeline/route';

      count: vi.fn()

      { id: 2, date: '1920年', title: '进步团体合影' }
    ];

    vi.mocked(prisma.timelineNode.count).mockResolvedValue(2);

    const request = new Request('http://localhost/api/v1/timeline');
    const response = await GET(request);
    const data = await response.json();

    expect(response.status).toBe(200);
    expect(data.data.items).toHaveLength(2);
    expect(data.data.total).toBe(2);
  });

  it('应支持era筛选', async () => {
    vi.mocked(prisma.timelineNode.findMany).mockResolvedValue([]);
    vi.mocked(prisma.timelineNode.count).mockResolvedValue(0);

    expect(prisma.timelineNode.findMany).toHaveBeenCalledWith(

        where: expect.objectContaining({
          eraId: 'yanan'

    );

    expect(data.data.pageSize).toBe(20);
```

```typescript
import { describe, it, expect, beforeAll, afterAll, beforeEach } from 'vitest';
import { prisma } from '@/lib/db';
import { TimelineService } from '@/server/services/timeline';

describe('TimelineService 集成测试', () => {
  beforeAll(async () => {
    // 连接测试数据库
    await prisma.$connect();
  });

  afterAll(async () => {
    await prisma.$disconnect();
  });

  beforeEach(async () => {

    await prisma.era.deleteMany();

    await prisma.era.createMany({
      data: [
        { id: 'youth', name: '求学探索时期', period: '1918-1920', sortOrder: 1 },
        { id: 'revolution', name: '革命征程时期', period: '1921-1936', sortOrder: 2 }
      ]
    });
  });

  it('应创建并查询时间节点', async () => {
    const service = new TimelineService();

    // 创建
    const created = await service.create({
      date: '1919年',
      dateSort: '1919',
      eraId: 'youth',
      title: '毛泽东在长沙',
      description: '毛泽东在长沙',
      thumbnailUrl: 'https://example.com/1.jpg',
      sortOrder: 1
    });

    expect(created.id).toBeDefined();

    // 查询
    const found = await service.findById(created.id);
    expect(found.title).toBe('毛泽东在长沙');
  });

  it('应更新时间节点', async () => {
    const service = new TimelineService();
    const created = await service.create({
      date: '1920年',
      dateSort: '1920',
      eraId: 'youth',
      title: '原标题',
      description: '原描述',
      thumbnailUrl: 'https://example.com/2.jpg',
      sortOrder: 2
    });

    const updated = await service.update(created.id, {
      title: '新标题',
      description: '新描述'
    });

    expect(updated.title).toBe('新标题');
    expect(updated.description).toBe('新描述');
  });

  it('应删除时间节点', async () => {
    const service = new TimelineService();
    const created = await service.create({
      date: '1921年',
      dateSort: '1921',
      eraId: 'revolution',
      title: '建党',
      description: '中国共产党成立',
      thumbnailUrl: 'https://example.com/3.jpg',
      sortOrder: 3
    });

    await service.delete(created.id);

    const found = await service.findById(created.id);
    expect(found).toBeNull();
  });

  it('应按年代筛选节点', async () => {
    const service = new TimelineService();
    await service.create({
      date: '1919年', dateSort: '1919', eraId: 'youth',
      title: '节点1', description: '描述1',
      thumbnailUrl: 'https://example.com/1.jpg', sortOrder: 1
    });

      date: '1921年', dateSort: '1921', eraId: 'revolution',
      title: '节点2', description: '描述2',
      thumbnailUrl: 'https://example.com/2.jpg', sortOrder: 2
    });

    const result = await service.findByEra('youth');
    expect(result.items).toHaveLength(1);
```

```typescript
import { describe, it, expect } from 'vitest';
import request from 'supertest';

const BASE_URL = 'http://localhost:3000/api/v1';

describe('时间轴 API', () => {
  it('GET /timeline 应返回节点列表', async () => {
    const res = await request(BASE_URL).get('/timeline');
    expect(res.status).toBe(200);
    expect(res.body.code).toBe(200);

    expect(res.body.data.total).toBeGreaterThan(0);
  });

  it('GET /timeline?era=yanan 应返回延安时期节点', async () => {
    const res = await request(BASE_URL).get('/timeline?era=yanan');
    expect(res.status).toBe(200);
    res.body.data.items.forEach((node: any) => {
      expect(node.eraId).toBe('yanan');
    });
  });

  it('GET /timeline?page=1&pageSize=10 应返回分页数据', async () => {
    const res = await request(BASE_URL).get('/timeline?page=1&pageSize=10');
    expect(res.status).toBe(200);
    expect(res.body.data.items.length).toBeLessThanOrEqual(10);
    expect(res.body.data.page).toBe(1);
    expect(res.body.data.pageSize).toBe(10);

  it('GET /timeline/{id} 应返回节点详情', async () => {
```

```javascript
import http from 'k6/http';
import { check, sleep } from 'k6';
import { Rate } from 'k6/metrics';

const errorRate = new Rate('errors');

export const options = {
  stages: [
    { duration: '30s', target: 50 },   // 30秒内增加到50并发
    { duration: '1m', target: 100 },   // 1分钟内增加到100并发
    { duration: '2m', target: 200 },   // 2分钟内增加到200并发
    { duration: '5m', target: 200 },   // 维持200并发5分钟
    { duration: '1m', target: 500 },   // 增加到500并发
    { duration: '2m', target: 500 },   // 维持500并发2分钟
    { duration: '30s', target: 0 },    // 降到0
  ],
  thresholds: {
    http_req_duration: ['p(95)<500', 'p(99)<1000'],
    errors: ['rate<0.01'],
  },
};

export default function () {
  // 测试时间轴列表
  const res = http.get('https://test.example.com/api/v1/timeline?page=1&pageSize=20');

  const checkRes = check(res, {
    'status is 200': (r) => r.status === 200,
    'response time < 500ms': (r) => r.timings.duration < 500,
    'has items': (r) => r.json('data.items') !== null,
  });

  errorRate.add(!checkRes);
  sleep(1);
}
```
