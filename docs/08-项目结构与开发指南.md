### 3.2 开发环境搭建

**前置要求**：
- Node.js 18 LTS+
- MySQL 8.0+
- Redis 7+
- npm / pnpm / yarn

**安装步骤**：
```bash
# 1. 克隆项目
git clone <repository-url>
cd mao-memorial-web

# 2. 安装依赖
npm install

# 3. 复制环境变量
cp .env.example .env.local
# 编辑 .env.local，填入实际的数据库密码等

# 4. 初始化数据库
npx prisma migrate dev --name init

# 5. 导入种子数据
npx prisma db seed

# 6. 导入时间线数据
npx tsx prisma/import-timeline.ts

# 7. 启动开发服务器
npm run dev
```

**访问地址**：
- 前台：http://localhost:3000
- 后台：http://localhost:3000/admin/login
- 默认管理员：admin / admin123

### 3.3 Docker 开发环境

```bash
# 使用 Docker Compose 一键启动
docker compose up -d

# 包含服务：
# - web (Next.js 应用)
# - mysql (数据库)
# - redis (缓存)
# - nginx (反向代理)
```

**`docker-compose.yml`**：
```yaml
version: '3.8'

services:
  mysql:
    image: mysql:8.0
    container_name: mao-mysql
    environment:
      MYSQL_ROOT_PASSWORD: ${MYSQL_ROOT_PASSWORD:-root123}
      MYSQL_DATABASE: mao_memorial
    ports:
      - "3306:3306"
    volumes:
      - mysql_data:/var/lib/mysql
    command: --character-set-server=utf8mb4 --collation-server=utf8mb4_unicode_ci

  redis:
    image: redis:7-alpine
    container_name: mao-redis
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data

  web:
    build: .
    container_name: mao-web
    ports:
      - "3000:3000"
    environment:
      - DATABASE_URL=mysql://root:${MYSQL_ROOT_PASSWORD:-root123}@mysql:3306/mao_memorial
      - REDIS_URL=redis://redis:6379
    depends_on:
      - mysql
      - redis
    volumes:
      - ./uploads:/data/uploads

  nginx:
    image: nginx:alpine
    container_name: mao-nginx
    ports:
      - "80:80"
    volumes:
      - ./nginx.conf:/etc/nginx/nginx.conf
    depends_on:
      - web

volumes:
  mysql_data:
  redis_data:
```

## 4. 开发规范

### 4.1 分支管理策略

采用 **Git Flow** 简化版：

| 分支 | 用途 | 命名规范 |
|------|------|---------|
| `main` | 生产环境 | - |
| `develop` | 开发集成 | - |

### 5.3 API 路由规范

```typescript
// src/app/api/v1/timeline/route.ts

import { NextRequest } from 'next/server';
import { prisma } from '@/lib/db';
import { apiSuccess, apiError } from '@/lib/api-response';
import { validateQuery } from '@/lib/validate';
import { z } from 'zod';

const querySchema = z.object({
  page: z.coerce.number().min(1).default(1),
  pageSize: z.coerce.number().min(1).max(50).default(10),
  era: z.string().optional(),
  year: z.coerce.number().optional(),
});

export async function GET(request: NextRequest) {
  try {
    const query = validateQuery(request, querySchema);

    const where = {
      isPublished: true,
      isDeleted: false,
      ...(query.era && { eraId: query.era }),
      ...(query.year && { year: query.year }),
    };

    const [items, total] = await Promise.all([
      prisma.timelineNode.findMany({
        where,
        orderBy: { dateSort: 'asc' },
        skip: (query.page - 1) * query.pageSize,
        take: query.pageSize,
        include: { photos: true },
      }),
      prisma.timelineNode.count({ where }),
    ]);

    return apiSuccess({
      items,
      total,
      page: query.page,
      pageSize: query.pageSize,
      totalPages: Math.ceil(total / query.pageSize),
    });
  } catch (error) {
    return apiError(error);
  }
}
```
