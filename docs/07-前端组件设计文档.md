## 1. 设计规范

### 1.1 设计主题

本项目以**庄重、肃穆、典雅**为设计基调，采用红色与金色为主色调，体现红色文化主题，同时保持现代感和良好的用户体验。

### 1.2 色彩系统

```css
/* 主色调 */
--color-primary: #8B0000;          /* 深红色 - 主色 */
--color-primary-dark: #5C0000;     /* 暗红色 - 悬停 */
--color-primary-light: #B22222;    /* 亮红色 - 辅助 */
--color-accent: #D4AF37;           /* 金色 - 强调色 */
--color-accent-light: #F0D060;     /* 浅金色 */

/* 中性色 */
--color-bg: #FAFAF7;               /* 米白背景 */
--color-bg-dark: #1A1A1A;          /* 深色背景 */
--color-surface: #FFFFFF;          /* 卡片背景 */
--color-text: #2C2C2C;             /* 主文字 */
--color-text-secondary: #666666;   /* 次要文字 */
--color-text-light: #999999;       /* 辅助文字 */
--color-border: #E5E5E5;           /* 边框 */

/* 年代主题色 */
--color-era-youth: #4A7C59;        /* 求学探索 - 墨绿 */
--color-era-revolution: #8B0000;   /* 革命征程 - 深红 */

/* 功能色 */

```

```tsx
export function TimelineNode({ node, index, isLast, onNodeClick }: TimelineNodeProps) {
  const ref = useRef<HTMLDivElement>(null);
```

```typescript
// src/hooks/use-timeline.ts
export function useTimeline(params: TimelineParams) {
  const { data, error, isLoading, mutate } = useSWR(
    ['/api/v1/timeline', params],
    ([url, p]) => fetcher(url, p),
    {
      revalidateOnFocus: false,
      dedupingInterval: 60000,
    }
  );
  return { data, error, isLoading, mutate };
}

// src/hooks/use-infinite-timeline.ts
export function useInfiniteTimeline(params: Omit<TimelineParams, 'page'>) {
  const { data, error, size, setSize, isLoading } = useSWRInfinite(
    (page) => ['/api/v1/timeline', { ...params, page: page + 1 }],
    ([url, p]) => fetcher(url, p),
    { revalidateFirstPage: false }
  );

  const nodes = data?.flatMap(d => d.items) ?? [];
  const hasMore = data?.[data.length - 1]?.page < data?.[data.length - 1]?.totalPages;

  return { nodes, error, isLoading, size, setSize, hasMore };
}
```
