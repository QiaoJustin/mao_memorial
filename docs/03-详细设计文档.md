## 1. 文档说明

### 1.1 文档目的

本文档在概要设计的基础上，对各功能模块进行详细设计，包括页面结构、组件设计、交互逻辑、数据流、核心算法等，作为开发人员编码实现的直接依据。

### 1.2 设计约定

- 组件命名：采用 PascalCase，如 `TimelineNode`
- 文件命名：页面采用 kebab-case 目录 + `page.tsx`，组件采用 PascalCase
- 状态变量：采用 camelCase
- 常量：采用 UPPER_SNAKE_CASE
- 类型定义：采用 PascalCase + Interface

### 3.3 懒加载与分页逻辑

**文件路径**：`src/hooks/useInfiniteTimeline.ts`

```typescript
import { useState, useEffect, useCallback } from 'react';

interface UseInfiniteTimelineOptions {
  pageSize: number;         // 每页数量
  era?: string;             // 年代筛选
}

export function useInfiniteTimeline({ pageSize, era }: UseInfiniteTimelineOptions) {
  const [nodes, setNodes] = useState<TimelineNodeData[]>([]);
  const [page, setPage] = useState(1);
  const [loading, setLoading] = useState(false);
  const [hasMore, setHasMore] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const loadMore = useCallback(async () => {
    if (loading || !hasMore) return;

    setLoading(true);
    setError(null);

    try {
      const params = new URLSearchParams({
        page: String(page),
        pageSize: String(pageSize),
      });
      if (era) params.set('era', era);

      const res = await fetch(`/api/v1/timeline?${params}`);
      const json = await res.json();

      if (json.code === 200) {
        const newNodes = json.data.items;
        setNodes(prev => [...prev, ...newNodes]);
        setHasMore(newNodes.length === pageSize);
        setPage(prev => prev + 1);
      } else {
        setError(json.message);
      }
    } catch (err) {
      setError('加载失败，请稍后重试');
    } finally {
      setLoading(false);
    }
  }, [page, pageSize, era, loading, hasMore]);

  // 重置（年代筛选变化时）
  useEffect(() => {
    setNodes([]);
    setPage(1);
    setHasMore(true);
    loadMore();
  }, [era]);

  return { nodes, loading, hasMore, error, loadMore };
}
```

### 3.4 滚动加载触发器

**文件路径**：`src/components/site/ScrollTrigger.tsx`

```typescript
'use client';

import { useEffect, useRef } from 'react';

interface ScrollTriggerProps {
  onTrigger: () => void;
  enabled: boolean;
}

export default function ScrollTrigger({ onTrigger, enabled }: ScrollTriggerProps) {
  const ref = useRef<HTMLDivElement>(null);

  useEffect(() => {
    if (!enabled) return;

    const observer = new IntersectionObserver(
      (entries) => {
        if (entries[0].isIntersecting) {
          onTrigger();
        }
      },
      { rootMargin: '200px' }
    );

    if (ref.current) {
      observer.observe(ref.current);
    }

    return () => observer.disconnect();
  }, [onTrigger, enabled]);

  return <div ref={ref} className="h-1" />;
}
```

```typescript
      const docHeight = document.documentElement.scrollHeight - window.innerHeight;
      const progressPercent = (scrollTop / docHeight) * 100;
      setProgress(progressPercent);

      // 根据滚动位置计算当前年份
      const viewportCenter = scrollTop + window.innerHeight / 2;
      const nodes = document.querySelectorAll('[data-node-date]');
      let closest = '';
      let closestDist = Infinity;

      nodes.forEach(node => {
        const rect = node.getBoundingClientRect();
        const nodeCenter = rect.top + scrollTop + rect.height / 2;
        const dist = Math.abs(nodeCenter - viewportCenter);
        if (dist < closestDist) {
          closestDist = dist;
          closest = node.getAttribute('data-node-date') || '';
        }
      });
      setCurrentYear(closest);
    };

    window.addEventListener('scroll', handleScroll, { passive: true });
    handleScroll();
    return () => window.removeEventListener('scroll', handleScroll);
  }, []);

  return (
    <div className="hidden lg:block fixed right-8 top-1/2 -translate-y-1/2 z-40">
      <div className="flex flex-col items-center gap-3">
        <div className="text-[#D4AF37] font-serif text-lg writing-vertical">
          {currentYear}
        </div>
        <div className="w-1 h-48 bg-[#333] rounded-full relative overflow-hidden">
          <div
            className="absolute top-0 left-0 w-full bg-gradient-to-b from-[#D4AF37] to-[#8B0000] transition-all duration-150"
            style={{ height: `${progress}%` }}
          />
        </div>
        <div className="text-gray-500 text-xs">
          {Math.round(progress)}%
        </div>
      </div>
    </div>
  );
}
```

### 4.2 灯箱组件 Lightbox

**文件路径**：`src/components/site/Lightbox.tsx`

**功能**：全屏图片查看，支持缩放、平移、切换。

```typescript
'use client';

import { motion, AnimatePresence } from 'framer-motion';
import { useState, useEffect, useRef } from 'react';
import Image from 'next/image';

interface LightboxProps {
  isOpen: boolean;
  photos: Photo[];
  currentIndex: number;
  onClose: () => void;
  onIndexChange: (index: number) => void;
}

export default function Lightbox({
  isOpen,
  photos,
  currentIndex,
  onClose,
  onIndexChange,
}: LightboxProps) {
  const [zoom, setZoom] = useState(1);
  const [position, setPosition] = useState({ x: 0, y: 0 });
  const isDragging = useRef(false);
  const dragStart = useRef({ x: 0, y: 0 });

  // 重置状态
  useEffect(() => {
    if (isOpen) {
      setZoom(1);
      setPosition({ x: 0, y: 0 });
    }
  }, [isOpen, currentIndex]);

  // 滚轮缩放
  const handleWheel = (e: WheelEvent) => {
    e.preventDefault();
    const delta = e.deltaY > 0 ? -0.1 : 0.1;
    setZoom(z => Math.min(Math.max(z + delta, 1), 4));
  };

  // 拖动平移
  const handleMouseDown = (e: React.MouseEvent) => {
    if (zoom === 1) return;
    isDragging.current = true;
    dragStart.current = {
      x: e.clientX - position.x,
      y: e.clientY - position.y,
    };
  };

  const handleMouseMove = (e: React.MouseEvent) => {
    if (!isDragging.current) return;
    setPosition({
      x: e.clientX - dragStart.current.x,
      y: e.clientY - dragStart.current.y,
    });
  };

  const handleMouseUp = () => {
    isDragging.current = false;
  };

  if (!isOpen || photos.length === 0) return null;

  return (
    <AnimatePresence>
      <motion.div
        initial={{ opacity: 0 }}
        animate={{ opacity: 1 }}
        exit={{ opacity: 0 }}
        className="fixed inset-0 z-[60] bg-black flex items-center justify-center"
        onClick={onClose}
      >
        {/* 关闭按钮 */}
        <button
          onClick={onClose}
          className="absolute top-6 right-6 w-12 h-12 flex items-center justify-center
                     text-white text-2xl hover:bg-white/10 rounded-full z-10"
        >
          ✕
        </button>

        {/* 图片 */}
        <div
          className="relative w-full h-full flex items-center justify-center overflow-hidden"
          onWheel={handleWheel}
          onMouseDown={handleMouseDown}
          onMouseMove={handleMouseMove}
          onMouseUp={handleMouseUp}
          onMouseLeave={handleMouseUp}
          onClick={e => e.stopPropagation()}
          style={{ cursor: zoom > 1 ? (isDragging.current ? 'grabbing' : 'grab') : 'default' }}
        >
          <motion.div
            animate={{
              scale: zoom,
              x: position.x,
              y: position.y,
            }}
            transition={{ type: 'spring', stiffness: 300, damping: 30 }}
            className="relative w-[80vw] h-[80vh]"
          >
            <Image
              src={photos[currentIndex].url}
              alt={photos[currentIndex].caption}
              fill
              className="object-contain"
              sizes="80vw"
              priority
            />
          </motion.div>
        </div>

        {/* 缩放控制 */}
        <div className="absolute bottom-6 left-1/2 -translate-x-1/2 flex items-center gap-4
                        bg-black/60 px-4 py-2 rounded-full">
          <button
            onClick={e => { e.stopPropagation(); setZoom(z => Math.max(z - 0.5, 1)); }}
            className="text-white text-xl w-8 h-8 hover:bg-white/10 rounded"
          >
            −
          </button>
          <span className="text-white text-sm w-12 text-center">
            {Math.round(zoom * 100)}%
          </span>
          <button
            onClick={e => { e.stopPropagation(); setZoom(z => Math.min(z + 0.5, 4)); }}
            className="text-white text-xl w-8 h-8 hover:bg-white/10 rounded"
          >
            +
          </button>
          <button
            onClick={e => {
              e.stopPropagation();
              setZoom(1);
              setPosition({ x: 0, y: 0 });
            }}
            className="text-white text-sm px-2 hover:bg-white/10 rounded"
          >
            重置
          </button>
        </div>

        {/* 左右切换 */}
        {photos.length > 1 && (
          <>
            <button
              onClick={e => {
                e.stopPropagation();
                onIndexChange((currentIndex - 1 + photos.length) % photos.length);
              }}
              className="absolute left-6 top-1/2 -translate-y-1/2 w-12 h-12
                         bg-white/10 hover:bg-white/20 rounded-full
                         flex items-center justify-center text-white text-2xl"
            >
              ←
            </button>
            <button
              onClick={e => {
                e.stopPropagation();
                onIndexChange((currentIndex + 1) % photos.length);
              }}
              className="absolute right-6 top-1/2 -translate-y-1/2 w-12 h-12
                         bg-white/10 hover:bg-white/20 rounded-full
                         flex items-center justify-center text-white text-2xl"
            >
              →
            </button>
          </>
        )}

        {/* 图片说明 */}
        <div className="absolute bottom-20 left-1/2 -translate-x-1/2 max-w-2xl px-4
                        text-center text-gray-300 text-sm">
          {photos[currentIndex].caption}
        </div>
      </motion.div>
    </AnimatePresence>
  );
}
```

```typescript
'use client';

import { useState, useEffect, useRef } from 'react';

interface SearchInputProps {
  onSearch: (keyword: string) => void;
  onSuggestionSelect: (keyword: string) => void;
}

export default function SearchInput({ onSearch, onSuggestionSelect }: SearchInputProps) {
  const [keyword, setKeyword] = useState('');
  const [suggestions, setSuggestions] = useState<string[]>([]);
  const [showSuggestions, setShowSuggestions] = useState(false);
  const debounceTimer = useRef<NodeJS.Timeout>();

  // 防抖搜索建议
  useEffect(() => {
    if (debounceTimer.current) clearTimeout(debounceTimer.current);

    if (keyword.trim().length < 2) {
      setSuggestions([]);
      return;
    }

    debounceTimer.current = setTimeout(async () => {
      try {
        const res = await fetch(`/api/v1/search/suggest?q=${encodeURIComponent(keyword)}`);
        const json = await res.json();
        if (json.code === 200) {
          setSuggestions(json.data.suggestions);
        }
      } catch (err) {
        console.error('获取搜索建议失败', err);
      }
    }, 300);

    return () => {
      if (debounceTimer.current) clearTimeout(debounceTimer.current);
    };
  }, [keyword]);

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    if (keyword.trim()) {
      onSearch(keyword.trim());
      setShowSuggestions(false);
    }
  };

  return (
    <div className="relative w-full max-w-2xl mx-auto">
      <form onSubmit={handleSubmit}>
        <div className="relative">
          <input
            type="text"
            value={keyword}
            onChange={e => setKeyword(e.target.value)}
            onFocus={() => setShowSuggestions(true)}
            onBlur={() => setTimeout(() => setShowSuggestions(false), 200)}
            placeholder="搜索时间节点、事件、关键词..."
            className="w-full px-6 py-4 pl-14 bg-[#1a1a1a] text-white
                       border border-[#D4AF37]/30 rounded-full
                       focus:outline-none focus:border-[#D4AF37]
                       transition-colors text-lg"
          />
          <span className="absolute left-5 top-1/2 -translate-y-1/2 text-[#D4AF37] text-xl">
            🔍
          </span>
          <button
            type="submit"
            className="absolute right-2 top-1/2 -translate-y-1/2
                       px-6 py-2 bg-[#8B0000] hover:bg-[#a00000]
                       text-white rounded-full transition-colors"
          >
            搜索
          </button>
        </div>
      </form>

      {/* 搜索建议 */}
      {showSuggestions && suggestions.length > 0 && (
        <div className="absolute top-full mt-2 w-full bg-[#1a1a1a]
                        border border-[#D4AF37]/20 rounded-lg shadow-2xl overflow-hidden z-20">
          {suggestions.map(s => (
            <button
              key={s}
              onClick={() => {
                setKeyword(s);
                onSuggestionSelect(s);
                setShowSuggestions(false);
              }}
              className="w-full px-6 py-3 text-left text-gray-300
                         hover:bg-white/5 transition-colors flex items-center gap-3"
            >
              <span className="text-[#D4AF37]">🔍</span>
              <span>{s}</span>
            </button>
          ))}
        </div>
      )}
    </div>
  );
}
```

```typescript
// src/lib/rate-limit.ts

import Redis from 'ioredis';

const redis = new Redis(process.env.REDIS_URL);

interface RateLimitResult {
  success: boolean;
  remaining: number;
  resetAt: number;
}

export async function checkLoginRateLimit(ip: string): Promise<RateLimitResult> {
  const key = `login_limit:${ip}`;
  const maxAttempts = 5;
  const windowSeconds = 1800; // 30分钟

  const attempts = await redis.incr(key);
  if (attempts === 1) {
    await redis.expire(key, windowSeconds);
  }

  const ttl = await redis.ttl(key);

  return {
    success: attempts <= maxAttempts,
    remaining: Math.max(0, maxAttempts - attempts),
    resetAt: Date.now() + ttl * 1000,
  };
}

export async function resetLoginRateLimit(ip: string): Promise<void> {
  await redis.del(`login_limit:${ip}`);
}
```

```typescript
/**
 * 解析中文日期字符串为排序用的日期对象
 * 支持："1949年"、"1949年10月"、"1949年10月1日"
 */
export function parseChineseDate(dateStr: string): Date {
  const match = dateStr.match(/(\d{4})年(\d{1,2})?月?(\d{1,2})?日?/);
  if (!match) return new Date(1900, 0, 1);

  const year = parseInt(match[1]);
  const month = match[2] ? parseInt(match[2]) - 1 : 0;
  const day = match[3] ? parseInt(match[3]) : 1;

  return new Date(year, month, day);
}

/**
 * 格式化日期为排序字符串
 */
export function formatDateSort(dateStr: string): string {
  const date = parseChineseDate(dateStr);
  const year = date.getFullYear();
  const month = String(date.getMonth() + 1).padStart(2, '0');
  const day = String(date.getDate()).padStart(2, '0');
  return `${year}-${month}-${day}`;
}
```

```typescript
interface SensitiveWord {
  word: string;
  level: number; // 1:替换 2:拒绝
}

let sensitiveWords: SensitiveWord[] = [];

// 初始化敏感词库
export async function initSensitiveWords(words: SensitiveWord[]) {
  sensitiveWords = words;
}

// 构建DFA敏感词树
function buildDFATree(words: string[]): Map<string, any> {
  const tree = new Map();
  for (const word of words) {
    let node = tree;
    for (const char of word) {
      if (!node.has(char)) {
        node.set(char, new Map());
      }
      node = node.get(char);
    }
    node.set('isEnd', true);
  }
  return tree;
}

// 检测敏感词
export function detectSensitiveWords(text: string): { found: boolean; words: string[] } {
  const found: string[] = [];
  for (const item of sensitiveWords) {
    if (text.includes(item.word)) {
      found.push(item.word);
    }
  }
  return { found: found.length > 0, words: found };
}

// 过滤敏感词（替换为*）
export function filterSensitiveWords(text: string): string {
  let result = text;
  for (const item of sensitiveWords) {
    if (item.level === 1) {
      const regex = new RegExp(item.word, 'gi');
      result = result.replace(regex, '*'.repeat(item.word.length));
    }
  }
  return result;
}
```

```typescript
/**
 * 规范化图片URL
 * 处理相对路径、协议缺失等情况
 */
export function normalizeImageUrl(url: string): string {
  if (!url) return '';
  if (url.startsWith('http://') || url.startsWith('https://')) {
    return url;
  }
  if (url.startsWith('//')) {
    return `https:${url}`;
  }
  if (url.startsWith('/')) {
    return `${process.env.NEXT_PUBLIC_CDN_BASE_URL || ''}${url}`;
  }
  return url;
}

/**
 * 生成图片缩略图URL
 * 利用CDN的图片处理功能
 */
export function getThumbnailUrl(url: string, width: number, height?: number): string {
  const normalized = normalizeImageUrl(url);
  // 阿里云OSS图片处理参数
  const params = `?x-oss-process=image/resize,w_${width}`;
  if (height) {
    return `${normalized}${params},h_${height}`;
  }
  return `${normalized}${params}`;
}

/**
 * 生成响应式图片srcset
 */
export function getResponsiveSrcSet(url: string): string {
  const widths = [480, 768, 1200, 1920];
  return widths
    .map(w => `${getThumbnailUrl(url, w)} ${w}w`)
    .join(', ');
}
```
