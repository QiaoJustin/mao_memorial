## 3. Docker 部署

### 3.1 Dockerfile

**文件**：`Dockerfile`

```dockerfile
# ============ 阶段1: 依赖安装 ============
FROM node:18-alpine AS deps
WORKDIR /app

COPY package.json package-lock.json ./
COPY prisma ./prisma/

RUN npm ci

# ============ 阶段2: 构建 ============
FROM node:18-alpine AS builder
WORKDIR /app

COPY --from=deps /app/node_modules ./node_modules
COPY . .

ENV NEXT_TELEMETRY_DISABLED=1
RUN npm run build

# ============ 阶段3: 运行 ============
FROM node:18-alpine AS runner
WORKDIR /app

ENV NODE_ENV=production
ENV NEXT_TELEMETRY_DISABLED=1

RUN addgroup --system --gid 1001 nodejs
RUN adduser --system --uid 1001 nextjs

COPY --from=builder /app/public ./public
COPY --from=builder --chown=nextjs:nodejs /app/.next/standalone ./
COPY --from=builder --chown=nextjs:nodejs /app/.next/static ./.next/static
COPY --from=builder /app/prisma ./prisma
COPY --from=builder /app/node_modules/.prisma ./node_modules/.prisma
COPY --from=builder /app/node_modules/@prisma ./node_modules/@prisma

USER nextjs

EXPOSE 3000

ENV PORT=3000
ENV HOSTNAME="0.0.0.0"

CMD ["node", "server.js"]
```

### 3.2 Docker Compose

**文件**：`docker-compose.yml`

```yaml
version: '3.8'

services:
  app:
    build:
      context: .
      dockerfile: Dockerfile
    container_name: mao-memorial-app
    restart: always
    ports:
      - "3000:3000"
    environment:
      - NODE_ENV=production
      - DATABASE_URL=mysql://mao_user:${DB_PASSWORD}@mysql:3306/mao_memorial
      - REDIS_URL=redis://redis:6379
      - JWT_SECRET=${JWT_SECRET}
      - NEXT_PUBLIC_BASE_URL=https://example.com
    depends_on:
      mysql:
        condition: service_healthy
      redis:
        condition: service_healthy
    networks:
      - mao-network

  mysql:
    image: mysql:8.0
    container_name: mao-memorial-mysql
    restart: always
    environment:
      - MYSQL_ROOT_PASSWORD=${MYSQL_ROOT_PASSWORD}
      - MYSQL_DATABASE=mao_memorial
      - MYSQL_USER=mao_user
      - MYSQL_PASSWORD=${DB_PASSWORD}
    volumes:
      - mysql-data:/var/lib/mysql
      - ./docker/mysql/conf.d:/etc/mysql/conf.d
      - ./docker/mysql/init:/docker-entrypoint-initdb.d
    ports:
      - "3306:3306"
    healthcheck:
      test: ["CMD", "mysqladmin", "ping", "-h", "localhost"]
      interval: 10s
      timeout: 5s
      retries: 5
    networks:
      - mao-network

  redis:
    image: redis:7-alpine
    container_name: mao-memorial-redis
    restart: always
    command: redis-server --requirepass ${REDIS_PASSWORD} --maxmemory 256mb --maxmemory-policy allkeys-lru
    volumes:
      - redis-data:/data
    ports:
      - "6379:6379"
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 5s
      retries: 5
    networks:
      - mao-network

  nginx:
    image: nginx:1.24-alpine
    container_name: mao-memorial-nginx
    restart: always
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./docker/nginx/conf.d:/etc/nginx/conf.d
      - ./docker/nginx/ssl:/etc/nginx/ssl
      - ./docker/nginx/logs:/var/log/nginx
    depends_on:
      - app
    networks:
      - mao-network

volumes:
  mysql-data:
  redis-data:

networks:
  mao-network:
    driver: bridge
```

## 4. Nginx 配置

### 4.1 主配置文件

**文件**：`/etc/nginx/nginx.conf`

```nginx
user www-data;
worker_processes auto;
pid /run/nginx.pid;
include /etc/nginx/modules-enabled/*.conf;

worker_rlimit_nofile 65535;

events {
    worker_connections 65535;
    multi_accept on;
}

http {
    # 基础配置
    sendfile on;
    tcp_nopush on;
    tcp_nodelay on;
    keepalive_timeout 65;
    types_hash_max_size 2048;
    server_tokens off;

    # MIME 类型
    include /etc/nginx/mime.types;
    default_type application/octet-stream;

    # 日志格式
    log_format main '$remote_addr - $remote_user [$time_local] '
                    '"$request" $status $body_bytes_sent '
                    '"$http_referer" "$http_user_agent" '
                    'rt=$request_time uct="$upstream_connect_time" '
                    'uht="$upstream_header_time" urt="$upstream_response_time"';

    access_log /var/log/nginx/access.log main;
    error_log /var/log/nginx/error.log warn;

    # Gzip 压缩
    gzip on;
    gzip_vary on;
    gzip_proxied any;
    gzip_comp_level 6;
    gzip_min_length 256;
    gzip_types
        text/plain
        text/css
        text/xml
        text/javascript
        application/javascript
        application/x-javascript
        application/json
        application/xml
        application/xml+rss
        application/rss+xml
        application/atom+xml
        image/svg+xml;

    # 缓冲区
    client_body_buffer_size 16K;
    client_header_buffer_size 1k;
    large_client_header_buffers 2 1k;
    client_max_body_size 20m;

    # 限流
    limit_req_zone $binary_remote_addr zone=api:10m rate=10r/s;
    limit_req_zone $binary_remote_addr zone=message:10m rate=1r/s;

    # 上游服务
    upstream nextjs_backend {
        server 127.0.0.1:3000 weight=5;
        # server 127.0.0.1:3001 weight=5;  # 多实例
        keepalive 32;
    }

    include /etc/nginx/conf.d/*.conf;
}
```

### 4.2 站点配置

**文件**：`/etc/nginx/conf.d/mao-memorial.conf`

```nginx
# HTTP -> HTTPS 重定向
server {
    listen 80;
    listen [::]:80;
    server_name example.com www.example.com;
    return 301 https://$host$request_uri;
}

# HTTPS 主配置
server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name example.com www.example.com;

    # SSL 证书
    ssl_certificate /etc/nginx/ssl/fullchain.pem;
    ssl_certificate_key /etc/nginx/ssl/privkey.pem;
    ssl_trusted_certificate /etc/nginx/ssl/chain.pem;

    # SSL 配置
    ssl_session_timeout 1d;
    ssl_session_cache shared:MozSSL:10m;
    ssl_session_tickets off;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384:ECDHE-ECDSA-CHACHA20-POLY1305:ECDHE-RSA-CHACHA20-POLY1305:DHE-RSA-AES128-GCM-SHA256:DHE-RSA-AES256-GCM-SHA384;
    ssl_prefer_server_ciphers off;

    # HSTS
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains; preload" always;

    # 安全头
    add_header X-Frame-Options "DENY" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;
    add_header Content-Security-Policy "default-src 'self'; script-src 'self' 'unsafe-inline' 'unsafe-eval'; style-src 'self' 'unsafe-inline'; img-src 'self' data: https:; font-src 'self' data:; connect-src 'self'; media-src 'self'; object-src 'none'; frame-ancestors 'none';" always;

    # 静态资源缓存
    location /_next/static/ {
        proxy_pass http://nextjs_backend;
        proxy_cache_valid 200 1y;
        add_header Cache-Control "public, immutable";
        expires 1y;
    }

    # 图片缓存
    location ~* \.(jpg|jpeg|png|gif|webp|svg|ico)$ {
        proxy_pass http://nextjs_backend;
        proxy_cache_valid 200 30d;
        add_header Cache-Control "public, max-age=2592000";
        expires 30d;
    }

    # API 限流
    location /api/ {
        limit_req zone=api burst=20 nodelay;
        proxy_pass http://nextjs_backend;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }

    # 留言接口额外限流
    location = /api/v1/messages {
        limit_req zone=message burst=3 nodelay;
        proxy_pass http://nextjs_backend;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # 后台接口仅允许特定IP
    location /api/v1/admin/ {
        # allow 10.0.0.0/8;      # 内网IP
        # allow 192.168.0.0/16;  # 内网IP
        # deny all;
        proxy_pass http://nextjs_backend;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # 主应用代理
    location / {
        proxy_pass http://nextjs_backend;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        proxy_read_timeout 86400;
    }

    # 健康检查
    location = /health {
        access_log off;
        return 200 "ok";
        add_header Content-Type text/plain;
    }
}
```

```bash
sudo cp privkey.pem /etc/nginx/ssl/
sudo cp chain.pem /etc/nginx/ssl/

# 设置权限
sudo chmod 644 /etc/nginx/ssl/fullchain.pem
sudo chmod 600 /etc/nginx/ssl/privkey.pem
sudo chmod 644 /etc/nginx/ssl/chain.pem

sudo nginx -t

# 重载 Nginx
sudo nginx -s reload
```

## 8. CI/CD 流水线

### 8.1 GitHub Actions 配置

**文件**：`.github/workflows/deploy.yml`

```yaml
name: Deploy to Production

on:
  push:
    branches: [main]
  workflow_dispatch:

env:
  NODE_VERSION: '18'
  
jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      
      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: ${{ env.NODE_VERSION }}
          cache: 'npm'
      
      - name: Install dependencies
        run: npm ci
      
      - name: Lint
        run: npm run lint
      
      - name: Unit tests
        run: npm run test:unit
      
      - name: Build
        run: npm run build
        env:
          DATABASE_URL: ${{ secrets.TEST_DATABASE_URL }}
      
      - name: E2E tests
        run: npm run test:e2e

  build-and-push:
    needs: test
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'
    steps:
      - uses: actions/checkout@v4
      
      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3
      
      - name: Login to Container Registry
        uses: docker/login-action@v3
        with:
          registry: ${{ secrets.REGISTRY_URL }}
          username: ${{ secrets.REGISTRY_USERNAME }}
          password: ${{ secrets.REGISTRY_PASSWORD }}
      
      - name: Build and push
        uses: docker/build-push-action@v5
        with:
          context: .
          push: true
          tags: |
            ${{ secrets.REGISTRY_URL }}/mao-memorial:latest
            ${{ secrets.REGISTRY_URL }}/mao-memorial:${{ github.sha }}
          cache-from: type=registry,ref=${{ secrets.REGISTRY_URL }}/mao-memorial:cache
          cache-to: type=registry,ref=${{ secrets.REGISTRY_URL }}/mao-memorial:cache,mode=max

  deploy:
    needs: build-and-push
    runs-on: ubuntu-latest
    environment: production
    steps:
      - name: Deploy to server
        uses: appleboy/ssh-action@v1.0.0
        with:
          host: ${{ secrets.SERVER_HOST }}
          username: ${{ secrets.SERVER_USER }}
          key: ${{ secrets.SERVER_SSH_KEY }}
          script: |
            cd /opt/mao-memorial
            docker compose pull
            docker compose --env-file .env.production up -d --build
            docker compose exec -T app npx prisma migrate deploy
            docker image prune -f
```

### 9.3 健康检查脚本

**文件**：`scripts/health-check.sh`

```bash
#!/bin/bash

# 健康检查脚本
URL="https://example.com/api/v1/health"
ALERT_EMAIL="admin@example.com"
MAX_RETRIES=3
RETRY_DELAY=5

check_health() {
  local response=$(curl -s -o /dev/null -w "%{http_code}" --max-time 10 "$URL")
  echo "$(date): Health check - HTTP $response"
  
  if [ "$response" != "200" ]; then
    return 1
  fi
  return 0
}

# 重试机制
for i in $(seq 1 $MAX_RETRIES); do
  if check_health; then
    echo "$(date): Health check passed"
    exit 0
  fi
  echo "$(date): Health check failed (attempt $i/$MAX_RETRIES)"
  sleep $RETRY_DELAY
done

# 所有重试失败，发送告警
echo "$(date): Health check failed after $MAX_RETRIES attempts. Sending alert."
# 发送告警邮件/短信/钉钉
# mail -s "ALERT: Mao Memorial Website is DOWN" $ALERT_EMAIL < /dev/null
exit 1
```

### 10.2 Winston 日志配置

**文件**：`src/lib/logger.ts`

```typescript
import winston from 'winston';
import path from 'path';

const logFormat = winston.format.combine(
  winston.format.timestamp({ format: 'YYYY-MM-DD HH:mm:ss' }),
  winston.format.errors({ stack: true }),
  winston.format.json()
);

const consoleFormat = winston.format.combine(
  winston.format.colorize(),
  winston.format.timestamp({ format: 'YYYY-MM-DD HH:mm:ss' }),
  winston.format.printf(({ timestamp, level, message, ...meta }) => {
    return `${timestamp} [${level}]: ${message} ${
      Object.keys(meta).length ? JSON.stringify(meta) : ''
    }`;
  })
);

export const logger = winston.createLogger({
  level: process.env.NODE_ENV === 'production' ? 'info' : 'debug',
  format: logFormat,
  defaultMeta: { service: 'mao-memorial' },
  transports: [
    // 错误日志文件
    new winston.transports.File({
      filename: path.join(process.env.LOG_DIR || 'logs', 'error.log'),
      level: 'error',
      maxsize: 50 * 1024 * 1024,  // 50MB
      maxFiles: 10,
    }),
    // 所有日志文件
    new winston.transports.File({
      filename: path.join(process.env.LOG_DIR || 'logs', 'combined.log'),
      maxsize: 50 * 1024 * 1024,
      maxFiles: 10,
    }),
    // 控制台输出（仅开发环境）
    ...(process.env.NODE_ENV !== 'production'
      ? [new winston.transports.Console({ format: consoleFormat })]
      : []),
  ],
});
```

## 11. 备份恢复

### 11.1 数据库备份脚本

**文件**：`scripts/backup-db.sh`

```bash
#!/bin/bash

# 配置
DB_HOST="127.0.0.1"
DB_USER="root"
DB_PASS="${MYSQL_ROOT_PASSWORD}"
DB_NAME="mao_memorial"
BACKUP_DIR="/data/backup/mysql"
DATE=$(date +%Y%m%d_%H%M%S)
BACKUP_FILE="${BACKUP_DIR}/${DB_NAME}_${DATE}.sql.gz"
RETENTION_DAYS=30

# 创建备份目录
mkdir -p ${BACKUP_DIR}

# 执行备份
echo "$(date): Starting database backup..."
mysqldump -h ${DB_HOST} -u ${DB_USER} -p${DB_PASS} \
  --single-transaction \
  --routines \
  --triggers \
  --quick \
  ${DB_NAME} | gzip > ${BACKUP_FILE}

# 检查备份结果
if [ $? -eq 0 ]; then
  FILESIZE=$(du -h ${BACKUP_FILE} | cut -f1)
  echo "$(date): Backup completed: ${BACKUP_FILE} (${FILESIZE})"
else
  echo "$(date): Backup FAILED!"
  # 发送告警
  exit 1
fi

# 清理旧备份
find ${BACKUP_DIR} -name "${DB_NAME}_*.sql.gz" -mtime +${RETENTION_DAYS} -delete
echo "$(date): Old backups cleaned (older than ${RETENTION_DAYS} days)"

# 可选：上传到 OSS
# ossutil cp ${BACKUP_FILE} oss://backup-bucket/mysql/
```

### 11.2 数据库恢复脚本

**文件**：`scripts/restore-db.sh`

```bash
#!/bin/bash

# 用法: ./restore-db.sh <backup_file.sql.gz>

if [ $# -ne 1 ]; then
  echo "Usage: $0 <backup_file.sql.gz>"
  exit 1
fi

BACKUP_FILE=$1
DB_HOST="127.0.0.1"
DB_USER="root"
DB_PASS="${MYSQL_ROOT_PASSWORD}"
DB_NAME="mao_memorial"

echo "$(date): Starting database restore from ${BACKUP_FILE}..."
echo "WARNING: This will overwrite the current database!"
read -p "Are you sure? (yes/no): " confirm
if [ "$confirm" != "yes" ]; then
  echo "Restore cancelled."
  exit 0
fi

# 解压并恢复
gunzip < ${BACKUP_FILE} | mysql -h ${DB_HOST} -u ${DB_USER} -p${DB_PASS} ${DB_NAME}

if [ $? -eq 0 ]; then
  echo "$(date): Restore completed successfully."
else
  echo "$(date): Restore FAILED!"
  exit 1
fi
```

## 12. 故障排查

### 12.1 常见故障及处理

| 故障现象 | 可能原因 | 排查步骤 | 解决方案 |
|---------|---------|---------|---------|
| 网站无法访问 | Nginx 停止 | `systemctl status nginx` | `systemctl restart nginx` |
| 网站无法访问 | 应用停止 | `pm2 status` | `pm2 restart mao-memorial` |
| 502 Bad Gateway | 应用端口未监听 | `netstat -tlnp \| grep 3000` | 重启应用 |
| 504 Gateway Timeout | 应用响应超时 | 查看应用日志 | 优化查询/增加超时 |
| 数据库连接失败 | MySQL 停止 | `systemctl status mysql` | `systemctl restart mysql` |
| 数据库连接失败 | 连接数满 | `SHOW STATUS LIKE 'Threads_connected'` | 增加连接数/排查泄漏 |
| Redis 连接失败 | Redis 停止 | `systemctl status redis` | `systemctl restart redis` |
| 图片加载失败 | OSS/CDN 故障 | 检查 OSS 状态 | 切换备用域名 |
| 页面加载慢 | 查询慢 | 检查慢查询日志 | 添加索引/优化 SQL |
| 内存溢出 | 内存泄漏 | `pm2 monit` | 重启应用/排查代码 |
